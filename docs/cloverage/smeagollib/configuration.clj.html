043    "The relative path to the config file."
044    (or
045      (env :smeagol-config)
046      (str (io/resource-path) "../config.edn")))

049  (defn- from-env-vars
050    "Read a map from those of these environment variables which have values"
051    [& vars]
052    (reduce
053      #(let [v (env %2)]
054         (if v (assoc %1 %2 v) %1))
055      {}
056      vars))

059  (defn to-keyword
060    "Convert this argument into an idiomatic clojure keyword."
061    [arg]
062    (if (and arg (not (keyword? arg)))
063      (keyword
064        (s/lower-case
065          (s/replace (str arg) #"[^A-Za-z0-9]+" "-")))
066      arg))

069  (defn transform-map
070    "transform this map `m` by applying these `transforms`. Each transforms
071    is expected to comprise a map with the keys :from and :to, whose values
072    are respectively a key to match and a key to replace that match with,
073    and optionally a key :transform, whose value is a function of one
074    argument to be used to transform the value of that key."
075    [m tuples]
076    (log/debug
077      "transform-map:\n"
078      (with-out-str (pprint m)))
079    (reduce
080      (fn [m tuple]
081        (if
082          (and (map? tuple) (map? m) (m (:from tuple)))
083          (let [old-val (m (:from tuple))
084                t (:transform tuple)]
085            (assoc
086              (dissoc m (:from tuple))
087              (:to tuple)
088              (if-not
089                (nil? t)
090                (eval (list t old-val)) old-val)))
091          m))
092      m
093      tuples))

096  (def config-env-transforms
097    "Transforms to use with `transform-map` to convert environment
098    variable names (which need to be specific) into the shorter names
099    used internally"
100    '( {:from :smeagol-content-dir :to :content-dir}
101       {:from :smeagol-default-locale :to :default-locale}
102       {:from :smeagol-formatters :to :formatters :transform read-string}
103       {:from :smeagol-js-from :to :extensions-from :transform to-keyword}
104       {:from :smeagol-log-level :to :log-level :transform to-keyword}
105       {:from :smeagol-passwd :to :passwd}
106       {:from :smeagol-site-title :to :site-title}))

109  (def build-config
110    "The actual configuration, as a map. The idea here is that the config
111    file is read (if it is specified and present), but that individual
112    values can be overridden by environment variables."
113    (memoize (fn []
114               (try
115                 (log/info (str "Reading configuration from " *config-file-path*))
116                 (let [file-contents (try
117                                       (read-string (slurp *config-file-path*))
118                                       (catch Exception x
119                                         (log/warn
120                                           (str
121                                             "Failed to read configuration from "
122                                             *config-file-path*
123                                             " because: "
124                                             (type x)
125                                             "; "
126                                             (.getMessage x)))
127                                         {}))
128                       config (merge
129                                file-contents
130                                (transform-map
131                                  (from-env-vars
132                                    :smeagol-content-dir
133                                    :smeagol-default-locale
134                                    :smeagol-formatters
135                                    :smeagol-js-from
136                                    :smeagol-log-level
137                                    :smeagol-passwd
138                                    :smeagol-site-title)
139                                  config-env-transforms))]
140                   (when (env :dev)
141                     (log/debug
142                       "Loaded configuration\n"
143                       (with-out-str (clojure.pprint/pprint config))))
144                   config)
145                 (catch Exception any
146                   (log/error any "Could not load configuration")
147                   {})))))
148  
149  (def config
150    "The actual configuration, as a map."
151    (build-config))

001  (ns ^{:doc "Functions related to the include of markdown-paged - parsing of
002  include links. This namespace is implementation detail for 
003  smeagollib.include and not inteded for direct usage."
004        :author "Michael Jerger"}
005    smeagollib.include.parse
006    (:require
007      [schema.core :as s]))
008  
009  (def IncludeLink
010    {:replace s/Str
011     :uri s/Str
012     :indent-heading s/Num
013     :indent-list s/Num})
014  
015  (s/defn
016    convert-indent-to-int :- s/Num
017    [indents :- [s/Str]]
018    (if (some? indents)
019        (Integer/valueOf (nth indents 2))
020        0))
021  
022  (s/defn
023    parse-indent-list
024    [md-src :- s/Str]
025    (re-matches #".*(:indent-list (\d)).*" md-src))
026  
027  (s/defn
028    parse-indent-heading
029    [md-src :- s/Str]
030    (re-matches #".*(:indent-heading (\d)).*" md-src))
031  
032  (s/defn
033    parse-include-link
034    [md-src :- s/Str]
035    (re-seq #".*(&\[\w*(.*)\w*\]\((.*)\)).*" md-src))
036  
037  (s/defn
038    parse-include-md :- [IncludeLink]
039    [md-src :- s/Str]
040    (vec
041      (map
042        (fn [parse-element]
043          (let [replace (nth parse-element 1)
044                uri (nth parse-element 3)
045                indents-text (nth parse-element 2)]
046            {:replace replace
047             :uri uri
048             :indent-heading (convert-indent-to-int (parse-indent-heading indents-text))
049             :indent-list (convert-indent-to-int (parse-indent-list indents-text))}))
050        (parse-include-link md-src))))

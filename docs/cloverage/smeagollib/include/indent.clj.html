001  (ns ^{:doc "Functions related to the include of markdown-paged - handling the 
002  list & heading indents of includes. This namespace is implementation detail for
003  smeagollib.include and not intended for direct usage."
004        :author "Michael Jerger"}
005    smeagollib.include.indent
006    (:require
007      [clojure.string :as cs]
008      [schema.core :as s]))
009  
010  (s/defn
011    parse-list
012    [md-resolved :- s/Str]
013    (distinct
014      (into
015        (re-seq #"((^|\R? *)([\*\+-] ))" md-resolved)
016        (re-seq  #"((^|\R? *)([0-9]+\. ))" md-resolved))))
017  
018  (s/defn
019    parse-heading
020    [md-resolved :- s/Str]
021    (distinct
022      (re-seq #"((^|\R?)(#+ ))" md-resolved)))
023  
024  (s/defn
025    do-indent :- s/Str
026    [indent :- s/Num
027     indentor :- s/Str
028     elements
029     md-resolved :- s/Str]
030    (loop [result md-resolved
031           elements elements]
032      (if (empty? elements)
033        result
034        (let [element (first elements)
035              replace (nth element 1)
036              start (nth element 2)
037              end (nth element 3)]
038          (recur
039            (cs/replace
040             result
041             (re-pattern (cs/escape
042                           replace
043                           {\* "\\*"
044                            \n "\\n"}))
045             (str start (apply str (repeat indent indentor)) end))
046            (rest elements))))))
047  
048  (s/defn
049    do-indent-heading :- s/Str
050    [indent :- s/Num
051     md-resolved :- s/Str]
052    (do-indent indent "#" (parse-heading md-resolved) md-resolved))
053  
054  (s/defn
055    do-indent-list :- s/Str
056    [indent :- s/Num
057     md-resolved :- s/Str]
058    (do-indent indent " " (parse-list md-resolved) md-resolved))

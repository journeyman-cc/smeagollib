057  (def local-url-base
058    "Essentially, the slash-terminated absolute path of the `public` resource
059    directory."
060    (let [a (str (fs/absolute content-dir))]
061      (subs a 0 (- (count a) (count "content")))))

063  (defn not-servable-reason
064    "As a string, the reason this `file-path` cannot safely be served, or `nil`
065    if it is safe to serve. This reason may be logged, but should *not* be
066    shown to remote users, as it would allow file system probing."
067    [file-path]
068    (try
069    (let [path (if
070                 (cs/starts-with? (str file-path) "/")
071                 file-path
072                 (cjio/file local-url-base file-path))]
073      (cond
074        (cs/includes? file-path "..")
075        (cs/join " " [file-path
076                      "Attempts to ascend the file hierarchy are disallowed."])
077        (not (cs/starts-with? path local-url-base))
078        (cs/join " " [path "is not servable"])
079        (not (fs/exists? path))
080        (cs/join " " [path "does not exist"])
081        (not (fs/readable? path))
082        (cs/join " " [path "is not readable"])))
083      (catch Exception any (cs/join " " [file-path "is not servable because" (.getMessage any)]))))

089  (defn local-url?
090    "True if this `file-path` can be served as a local URL, else false."
091    [file-path]
092    (try
093      (if
094        (empty? (not-servable-reason file-path))
095        true
096        (do
097          (log/error
098            "In `smeagollib.util/local-url? `" file-path "` is not a servable resource.")
099          false))
100      (catch Exception any
101        (log/error
102          "In `smeagollib.util/local-url `" file-path "` is not a servable resource:" any)
103        false)))

105  (defn local-url
106    "Return a local URL for this `file-path`, or a deliberate 404 if none
107    can be safely served."
108    ;; TODO: this actually returns a relative URL relative to local-url-base.
109    ;; That's not quite what we want because in Tomcat contexts the absolute
110    ;; URL may be different. We *ought* to be able to extract the offset from the
111    ;; servlet context, but it may be simpler to jam it in the config.
112    [file-path]
113    (try
114      (let [path (if
115                   (cs/starts-with? file-path local-url-base)
116                   (subs file-path (count local-url-base))
117                   file-path)
118            problem (not-servable-reason path)]
119        (if
120          (empty? problem)
121          path
122          (do
123            (log/error
124              "In `smeagollib.util/local-url `" file-path "` is not a servable resource.")
125            file-path)))
126      (catch Exception any
127        (log/error
128          "In `smeagollib.util/local-url `" file-path "` is not a servable resource:" any)
129        file-path)))

152  (defn get-servlet-context-path
153    "Return the servlet context path, if we're running as a servlet; if
154     not, return `nil`."
155    [request]
156    (when-let [context (:servlet-context request)]
157      ;; If we're not inside a serlvet environment (for
158      ;; example when using mock requests), then
159      ;; .getContextPath might not exist
160      (try
161        (let [path (.getContextPath context)
162              path' (if-not (cs/ends-with? path "/")
163                      (str path "/")
164                      path)]
165          path')
166        (catch IllegalArgumentException _ context))))

169  (def get-messages
170    "Return the most acceptable messages collection we have given the
171    `Accept-Language` header in this `request`."
172    (memoize
173      (fn [request]
174        (let [specifier ((:headers request) "accept-language")
175              messages (try
176                         (i18n/get-messages specifier "i18n" "en-GB")
177                         (catch Exception any
178                           (log/error
179                             any
180                             (str
181                               "Failed to parse accept-language header '"
182                               specifier
183                               "'"))
184                           {}))]
185      (merge
186        messages
187        config)))))

190  (defn get-message
191    "Return the message with this `message-key` from this `request`.
192    if not found, return this `default`, if provided; else return the
193    `message-key`."
194    ([message-key request]
195     (get-message message-key message-key request))
196    ([message-key default request]
197     (let [messages (get-messages request)]
198       (if
199         (map? messages)
200         (or (messages message-key) default)
201         default))))

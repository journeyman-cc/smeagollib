041  ;; The general idea behind the 'check-' functions in this file is that, if the
042  ;; check passes, they return true; if it fails, they return a map of problems found.
043  ;; the map comprises keys bound to 'explanation' lists of keywords and strings. If
044  ;; internationalisation is available, the keywords will then be translated into
045  ;; localised strings for presentation to the user; but if it isn't available,
046  ;; the keywords need to be human readable. Sanity checking ought to work even
047  ;; when the installation is quite badly broken.
048  
049  (defn check-exists
050    "Check this `path` exists. If so, return `true`; if not, return a map
051    containing this `problem-key` bound to a list explaining the problem."
052    [path problem-key]
053    (if-not
054      (.exists (cjio/as-file path))
055      {problem-key (list :file-or-directory path :does-not-exist)}
056      true))

059  (defn check-is-dir
060    "Check this `path` is a directory. If so, return `true`; if not, return a map
061    containing this `problem-key` bound to a list explaining the problem."
062    [path problem-key]
063    (if-not
064      (.isDirectory (cjio/as-file path))
065      {problem-key (list :file-or-directory path :is-not-directory)}
066      true))

089  (defn check-with-protection
090    "Apply this `check` to this `path` and `problem-key`; if no exception is thrown, return
091    the result. If an exception is thrown, return a map comprising a problem-key bound to
092    an explanation which includes the exception."
093    [check problem-key & args]
094    (try
095      (apply check args)
096      (catch Exception ex
097        {problem-key (list problem-key args ex)})))

100  (defn compound-check-results
101    [& results]
102    (let [problems (remove true? results)]
103      (if (empty? problems) true (apply merge problems))))

106  (defn check-can-read-and-write
107    "Check this `path` is both readable and writable. If so, return `true`;
108    if not, return a map containing this `problem-key` bound to a list explaining
109    the problem."
110    [path problem-key]
111    (compound-check-results
112      (check-with-protection check-exists :file-or-directory path (keyword (str (name problem-key) "-exists")))
113      (check-with-protection check-can-read :file-or-directory path (keyword (str (name problem-key) "-can-read")))
114      (check-with-protection check-can-write :file-or-directory path (keyword (str (name problem-key) "-can-write")))))

117  (defn check-content-dir
118    "Check that `path` exists and is populated as a valid content directory. Return true
119    if so, else a map of all problems found. If `path` is not supplied, default to the
120    configured content directory."
121    ([path]
122    (compound-check-results
123      (check-with-protection check-is-dir :file-or-directory path :content-dir-is-dir)
124      (check-can-read-and-write path :content-dir)
125      (apply compound-check-results
126        (map
127          #(check-can-read-and-write
128             (cjio/file path (str "_" % ".md"))
129             %)
130          ["side-bar" "edit-side-bar" "header" ]))))
131    ([]
132     (check-content-dir content-dir)))

135  (defn check-upload-dir
136    "Check that the upload directory at `path` (defaults to configured upload path)
137    exists and is readable and writable; check that each of the configured
138    thumbnail subdirectories also exists and is writable."
139    ([path]
140     (compound-check-results
141       (check-with-protection check-is-dir :file-or-directory path :upload-dir-is-dir)
142       (check-can-read-and-write path :upload-dir)
143       (apply
144         compound-check-results
145         (map
146           #(let [p (cjio/file path (name %))]
147              (compound-check-results
148                (check-with-protection check-is-dir :file-or-directory p :thumbnails)
149                (check-with-protection check-can-read-and-write :file-or-directory p :thumbnails)))
150           (keys (:thumbnails config))))))
151    ([] (check-upload-dir upload-dir)))

154  (defn check-password-member-field
155    "Check that this `member` map, expected to be an entry from the passwd
156    file whose key was `user-key`, has this `field` and if not return a
157    problem explanation with this `problem-key`."
158    [member field user-key problem-key]
159    (if
160      (and (map? member) (member field))
161      true
162      {problem-key (list :user-lacks-field user-key field)}))

166    "Check that this `member` map, expected to be an entry from the passwd
167    file whose key was `user-key`, has all the required fields and if not
168    return a problem explanation with this `problem-key`."
169    [member user-key problem-key]
170    (apply

172      (map
173        #(check-password-member-field
174           member

184        [:email :password])))

223    "Check that the file at this `path` is a valid configuration file"
224    [path]
225    (let [content (try
226                    (read-string (slurp path))
227                    (catch Exception _ {}))]
228      (compound-check-results

231        (if-not
232          (:site-title content)

234          true)

236          (:default-locale content)
237          {:default-locale-not-configured :default-locale-not-configured}
238          true))))

261  ;; ExplanationPart is a protocol for ensuring that everything which may form part of a
262  ;; problem explanation can be formatted into hiccup, so that it can be converted by
263  ;; hiccup into HTML. The reason for using Hiccup rather than Selmer is that in
264  ;; sanity check I don't want to be dependent on the existance of templates.
265  ;; (Also, I personally like Hiccup better, although I know it's too geeky for most
266  ;; people)
267  (defprotocol ExplanationPart
268    "things which may be parts of explanations need mechanisms for reducing
269    themselves to natural language where possible"
270    (as-hiccup [this dictionary] "Return `this` as a hiccup-formatted structure."))
271  
272  (extend-protocol ExplanationPart
273    nil
274    (as-hiccup [this dictionary] "")

282                 " "))

432  (defn- raw-sanity-check-installation
433    "Actually do the sanity check."
434    []
435    (timbre/info "Running sanity check")
436    (let [result (check-everything)]
437      (if
438        (map? result)
439        (do
440          (timbre/warn "Sanity check completed; " (count (keys result)) " problem(s) found")
441          (sanity-check-report result))
442        (do
443          (timbre/info "Sanity check completed; no problems found")
444          nil))))

452  (defn show-sanity-check-error
453    "Generate an error page in a way which should work even when everything else is broken.
454    If no argument is passed, run the sanity check and if it fails return page contents;
455    if `error` is passed, just return page content describing the error."
456    ([error]
457     (let [messages (get-locale-messages)]
458       (html
459         [:html
460          (as-hiccup-head messages)
461          [:body
462           (as-hiccup-header messages)
463           [:div {:id "error"}
464            [:p {:class "error"} (.getMessage error)]]
465           [:div {:id "main-container" :class "sanity-check-report"}
466            [:p  (as-hiccup :smeagol-misconfiguration messages)]
467            (as-hiccup error messages)
468            (as-hiccup-see-doc messages)]
469           (as-hiccup-footer messages)]])))
470    ([]
471     (try
472       (sanity-check-installation)
473       (catch Exception any
474         (timbre/error any "Failure during sanity check")
475         (show-sanity-check-error any)))))

063  (defn process-vega
064    "If this `src-resource-or-url` is a valid URL, it is assumed to point to a
065    plain text file pointing to  valid `vega-src`; otherwise, it is expected to
066    BE a valid `vega-src`.
067  
068    Process this `vega-src` string, assumed to be in YAML format, into a
069    specification of a Vega chart, and add the plumbing to render it."
070    [^String src-resource-or-url ^Integer index]
071    (let [data (resource-url-or-data->data src-resource-or-url)
072          vega-src (:data data)]
073      (log/info "Retrieved vega-src from " (:from data) " `" ((:from data) data) "`")
074      (str
075        "<div class='data-visualisation' id='vis" index "'></div>\n"
076        "<script>\n//<![CDATA[\nvar vl"
077        index
078        " = "
079        (yaml->json (str "$schema: https://vega.github.io/schema/vega-lite/v2.json\n" vega-src))
080        ";\nvegaEmbed('#vis"
081        index
082        "', vl"
083        index
084        ");\n//]]\n</script>")))

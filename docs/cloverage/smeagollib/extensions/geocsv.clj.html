031  (defn process-geocsv
032    "If this `url-or-geo-csv` is a valid URL, it is assumed to point to a CSV file
033    containing geographical point data; otherwise, it is expected to be CSV formatted
034    text with at least `latitude` and `longitude` columns."
035    [^String url-or-geo-csv ^Integer index]
036    (let [data (resource-url-or-data->data url-or-geo-csv)
037          geo-csv (:data data)]
038      (log/info "Retrieved geo-csv from " (:from data) " `" ((:from data) data) "`")
039      (str "\n<div class=\"geocsv\" style=\"height: 600px;\" id=\"geocsv-" index
040           "\">\n<pre>\n"
041           geo-csv
042           "\n</pre>
043           </div>
044           <script>
045             //<![CDATA[
046               document.onreadystatechange = function () {
047                 if (document.readyState === 'interactive') {
048                   GeoCSV.setIconUrlBase( \""
049                   (-> config :formatters :geocsv :icon-url-base) "\");
050                   GeoCSV.initialiseMapElement(\"geocsv-" index "\",
051                   document.getElementById(\"geocsv-" index "\").innerText.trim().replace(/\\[\\[([^\\[\\]]*)\\]\\]/g, \"<a href='wiki?page=$1'>$1</a>\"));
052                  }};
053             //]]
054           </script>
055           ")))

036  (defn local-links
037    "Rewrite text in `html-src` surrounded by double square brackets as a local link into this wiki."
038    [^String html-src]
039    (if html-src
040      (cs/replace html-src #"\[\[[^\[\]]*\]\]"
041                  #(let [text (cs/replace %1 #"[\[\]]" "")
042                         encoded (url-encode text)
043                         ;; I use '\_' to represent '_' in wiki markup, because
044                         ;; '_' is meaningful in Markdown. However, this needs to
045                         ;; be stripped out when interpreting local links.
046                         munged (cs/replace encoded #"%26%2395%3B" "_")]
047                     (format "<a href='wiki?page=%s'>%s</a>" munged text)))
048      no-text-error))
